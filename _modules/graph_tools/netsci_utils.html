[docs]def plot_all_triads(
    order=nsm.triad_order_nn4576,
    o=(1, 0), delta=(1, 0),
    r=.4,
    node_colors = None,
    **kwargs):
    
    return_list = []
    if node_colors is None:
        node_colors = [None]*len(order)
    
    for i,col in zip(range(len(order)),node_colors):
        return_list.append(
            nsv.plot_a_triad(
                order[i], 
                o=np.array(o) + np.array([i * delta[0], np.mod(i, 2) * delta[1]]),
                r=r, 
                node_color = col,
                **kwargs)
        )
            
    plt.axis('equal')
